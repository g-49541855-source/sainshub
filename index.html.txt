    <!-- Firebase SDK -->
    <script type="module">
        import { initializeApp } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-app.js";
        import { getAuth, signInAnonymously, signInWithCustomToken, onAuthStateChanged } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-auth.js";
        import { getFirestore, collection, addDoc, onSnapshot, deleteDoc, doc, query } from "https://www.gstatic.com/firebasejs/11.6.1/firebase-firestore.js";

        // Konfigurasi Baharu
        const manualConfig = {
            apiKey: "AIzaSyA5ZmdGcvIkc8XC3uIGDS4jWgMO7qP1dXQ",
            authDomain: "sains-hub.firebaseapp.com",
            databaseURL: "https://sains-hub-default-rtdb.asia-southeast1.firebasedatabase.app",
            projectId: "sains-hub",
            storageBucket: "sains-hub.firebasestorage.app",
            messagingSenderId: "714965517236",
            appId: "1:714965517236:web:aae454b019cf6983ea28e1"
        };

        const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : manualConfig;
        const app = initializeApp(firebaseConfig);
        const auth = getAuth(app);
        const db = getFirestore(app);
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'panitia-sains-pro';

        // Negeri Global
        let isAdmin = localStorage.getItem('panitia_admin') === 'true';
        let currentCollection = '';
        let teachers = [];
        let unsubscribeCurrent = null;

        // Bantuan UI
        window.switchTab = (tabId, colName = '', title = '') => {
            if (unsubscribeCurrent) {
                unsubscribeCurrent();
                unsubscribeCurrent = null;
            }

            document.querySelectorAll('.tab-content').forEach(t => t.classList.remove('active'));
            
            if (tabId === 'home') {
                document.getElementById('home').classList.add('active');
            } else if (tabId === 'activities') {
                document.getElementById('gallery-page').classList.add('active');
                if (auth.currentUser) fetchGallery();
            } else {
                currentCollection = colName;
                document.getElementById('page-title').innerText = title;
                document.getElementById('resource-page').classList.add('active');
                if (auth.currentUser) fetchResources(colName);
            }
            window.scrollTo(0, 0);
        };

        window.openModal = (id) => document.getElementById(id).classList.remove('hidden');
        window.closeModal = (id) => document.getElementById(id).classList.add('hidden');
        window.toggleAddForm = () => document.getElementById('admin-form').classList.toggle('hidden');

        // Logik Pengesahan
        window.handleLogin = (e) => {
            e.preventDefault();
            const u = document.getElementById('login-user').value;
            const p = document.getElementById('login-pass').value;
            if (u === 'admin' && p === 'sains123') {
                isAdmin = true;
                localStorage.setItem('panitia_admin', 'true');
                updateAdminUI();
                closeModal('login-modal');
                const activeTab = document.querySelector('.tab-content.active').id;
                if (activeTab === 'resource-page') fetchResources(currentCollection);
                if (activeTab === 'gallery-page') fetchGallery();
            } else {
                alert('Akses Ditolak!');
            }
        };

        window.logout = () => {
            isAdmin = false;
            localStorage.removeItem('panitia_admin');
            updateAdminUI();
            location.reload();
        };

        const updateAdminUI = () => {
            const loginBtn = document.getElementById('login-btn');
            const logoutBtn = document.getElementById('logout-btn');
            const addBtn = document.getElementById('add-btn');
            const galleryControls = document.getElementById('gallery-admin-controls');

            if (isAdmin) {
                loginBtn?.classList.add('hidden');
                logoutBtn?.classList.remove('hidden');
                addBtn?.classList.remove('hidden');
                galleryControls?.classList.remove('hidden');
            } else {
                loginBtn?.classList.remove('hidden');
                logoutBtn?.classList.add('hidden');
                addBtn?.classList.add('hidden');
                galleryControls?.classList.add('hidden');
            }
        };

        // Pemampat Imej
        const compressImage = (file) => {
            return new Promise((resolve) => {
                const reader = new FileReader();
                reader.readAsDataURL(file);
                reader.onload = (e) => {
                    const img = new Image();
                    img.src = e.target.result;
                    img.onload = () => {
                        const canvas = document.createElement('canvas');
                        const MAX_SIZE = 800;
                        let width = img.width;
                        let height = img.height;

                        if (width > height) {
                            if (width > MAX_SIZE) { height *= MAX_SIZE / width; width = MAX_SIZE; }
                        } else {
                            if (height > MAX_SIZE) { width *= MAX_SIZE / height; height = MAX_SIZE; }
                        }

                        canvas.width = width;
                        canvas.height = height;
                        const ctx = canvas.getContext('2d');
                        ctx.drawImage(img, 0, 0, width, height);
                        resolve(canvas.toDataURL('image/jpeg', 0.7));
                    };
                };
            });
        };

        // Operasi Firestore
        window.handleSubmit = async (e) => {
            e.preventDefault();
            if (!auth.currentUser) return;
            const title = document.getElementById('form-title').value;
            const link = document.getElementById('form-link').value;
            
            try {
                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', currentCollection), {
                    title, link, createdAt: new Date().toISOString()
                });
                document.getElementById('admin-form').classList.add('hidden');
                e.target.reset();
            } catch (err) { alert('Gagal menyimpan.'); }
        };

        window.handleFileUpload = async (e) => {
            const file = e.target.files[0];
            if (!file || !auth.currentUser) return;

            const caption = prompt("Masukkan tajuk gambar:");
            if (!caption) return;

            try {
                const compressed = await compressImage(file);
                await addDoc(collection(db, 'artifacts', appId, 'public', 'data', 'activities'), {
                    title: caption, link: compressed, createdAt: new Date().toISOString()
                });
            } catch (err) { alert('Muat naik gagal.'); }
        };

        window.deleteItem = async (col, id) => {
            if (!auth.currentUser) return;
            if (confirm('Padam item ini?')) {
                await deleteDoc(doc(db, 'artifacts', appId, 'public', 'data', col, id));
            }
        };

        // Pengambilan Data
        const fetchResources = (colName) => {
            const listDiv = document.getElementById('resource-list');
            listDiv.innerHTML = '<div class="text-center py-10 opacity-50">Menunggu data...</div>';
            
            if (unsubscribeCurrent) unsubscribeCurrent();
            
            unsubscribeCurrent = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', colName), (snapshot) => {
                listDiv.innerHTML = '';
                if (snapshot.empty) {
                    listDiv.innerHTML = '<div class="text-center py-20 glass rounded-3xl opacity-30 italic">Tiada data dijumpai.</div>';
                    return;
                }
                snapshot.docs.forEach(d => {
                    const data = d.data();
                    const item = document.createElement('div');
                    item.className = "flex items-center justify-between p-5 glass border border-white/5 rounded-2xl hover:bg-white/10 transition-all";
                    item.innerHTML = `
                        <div class="flex items-center gap-4">
                            <div class="p-3 bg-cyan-500/10 rounded-xl text-cyan-400"><i class="fas fa-file-alt"></i></div>
                            <h4 class="text-white font-bold">${data.title}</h4>
                        </div>
                        <div class="flex items-center gap-2">
                            <a href="${data.link}" target="_blank" class="px-4 py-2 bg-cyan-500 text-black text-xs font-black rounded-lg uppercase">Buka</a>
                            ${isAdmin ? `<button onclick="deleteItem('${colName}', '${d.id}')" class="p-2 text-red-400 hover:bg-red-500/10 rounded-lg"><i class="fas fa-trash"></i></button>` : ''}
                        </div>
                    `;
                    listDiv.appendChild(item);
                });
            }, (error) => {
                console.error("Firestore Error:", error);
                listDiv.innerHTML = '<div class="text-center py-10 text-red-400">Ralat akses data. Sila cuba sebentar lagi.</div>';
            });
        };

        const fetchGallery = () => {
            const grid = document.getElementById('gallery-grid');
            if (unsubscribeCurrent) unsubscribeCurrent();

            unsubscribeCurrent = onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'activities'), (snapshot) => {
                grid.innerHTML = '';
                snapshot.docs.forEach(d => {
                    const data = d.data();
                    const card = document.createElement('div');
                    card.className = "relative aspect-square rounded-[2rem] overflow-hidden glass group";
                    card.innerHTML = `
                        <img src="${data.link}" class="w-full h-full object-cover group-hover:scale-110 transition-transform duration-700">
                        <div class="absolute inset-0 bg-gradient-to-t from-black via-transparent to-transparent opacity-0 group-hover:opacity-100 flex items-end p-6 transition-opacity">
                            <p class="text-white text-[10px] font-bold uppercase tracking-widest">${data.title}</p>
                        </div>
                        ${isAdmin ? `<button onclick="deleteItem('activities', '${d.id}')" class="absolute top-4 right-4 bg-red-500 p-2 rounded-xl text-white"><i class="fas fa-trash-alt"></i></button>` : ''}
                    `;
                    grid.appendChild(card);
                });
            });
        };

        // Inisialisasi Menu & Slider
        const menuItems = [
            { id: 'teachers', label: 'Warga Panitia', icon: 'fa-users', col: 'teachers' },
            { id: 'dskp', label: 'DSKP & RPT', icon: 'fa-folder-open', col: 'dskp' },
            { id: 'reports', label: 'Laporan Aktiviti', icon: 'fa-file-signature', col: 'reports' },
            { id: 'minutes', label: 'Borang & Minit', icon: 'fa-clipboard-list', col: 'minutes' },
            { id: 'pbd', label: 'Rekod PBD', icon: 'fa-chart-pie', col: 'pbd' },
            { id: 'plc', label: 'Rekod PLC', icon: 'fa-chalkboard-teacher', col: 'plc' },
            { id: 'exams', label: 'Bank Soalan', icon: 'fa-book-open', col: 'exams' },
            { id: 'gallery', label: 'Galeri Aktiviti', icon: 'fa-images', col: 'activities' }
        ];

        const initUI = () => {
            const menu = document.getElementById('main-menu');
            menu.innerHTML = '';
            menuItems.forEach(item => {
                const btn = document.createElement('button');
                btn.className = "p-6 glass border border-white/5 rounded-[2rem] hover:bg-cyan-500/10 transition-all flex flex-col items-center gap-3 group";
                btn.onclick = () => switchTab(item.id === 'gallery' ? 'activities' : 'resource', item.col, item.label);
                btn.innerHTML = `
                    <div class="p-3 bg-white/5 rounded-2xl group-hover:scale-110 transition-transform"><i class="fas ${item.icon} text-cyan-400 text-2xl"></i></div>
                    <p class="text-[10px] font-black uppercase text-gray-400 group-hover:text-white tracking-widest text-center">${item.label}</p>
                `;
                menu.appendChild(btn);
            });
            updateAdminUI();
        };

        let slideIdx = 0;
        let sliderInterval = null;
        const startSlider = () => {
            const slider = document.getElementById('teacher-slider');
            if (teachers.length === 0) {
                slider.innerHTML = '<div class="flex items-center justify-center h-full text-gray-500 italic p-10 text-center">Tiada data guru.</div>';
                return;
            }

            if (sliderInterval) clearInterval(sliderInterval);

            const updateSlide = () => {
                const t = teachers[slideIdx];
                slider.innerHTML = `
                    <img src="${t.link || 'https://placehold.co/600x800/112240/06b6d4?text=Sains'}" class="w-full h-full object-cover animate-fade-in">
                    <div class="absolute inset-0 bg-gradient-to-t from-[#0a192f] via-transparent to-transparent"></div>
                    <div class="absolute bottom-0 p-10">
                        <span class="px-3 py-1 bg-cyan-500 text-black text-[10px] font-black uppercase rounded-lg mb-2 inline-block">${t.role || 'GURU'}</span>
                        <h4 class="text-white text-3xl font-black">${t.title}</h4>
                    </div>
                `;
                slideIdx = (slideIdx + 1) % teachers.length;
            };
            updateSlide();
            sliderInterval = setInterval(updateSlide, 5000);
        };

        const startApp = async () => {
            initUI();
            
            try {
                if (typeof __initial_auth_token !== 'undefined' && __initial_auth_token) {
                    await signInWithCustomToken(auth, __initial_auth_token);
                } else {
                    await signInAnonymously(auth);
                }
            } catch (err) {
                console.error("Auth Error:", err);
            }

            onAuthStateChanged(auth, (user) => {
                if (user) {
                    onSnapshot(collection(db, 'artifacts', appId, 'public', 'data', 'teachers'), (s) => {
                        teachers = s.docs.map(d => ({id: d.id, ...d.data()}));
                        startSlider();
                    }, (err) => console.error("Teachers Snapshot Error:", err));
                }
            });
        };

        window.onload = startApp;
    </script>
